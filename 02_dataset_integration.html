1.2 Diagnóstico de fiabilidad de sensores de tráfico¶

Los sensores de los tramos (Itineraris) pueden fallar de dos maneras principales:

  • Corte total — el sensor deja de transmitir → aparecen NaN en factor_medio. Se detecta como porcentaje de nulos > 20 % en un mes × tram.
  • Lecturas anómalas — el sensor transmite pero los valores no son coherentes con el comportamiento histórico del propio tramo o con sus vecinos del mismo corredor.

Se aplican dos modelos complementarios para detectar lecturas anómalas:

  1. Isolation Forest por tramo — aprende el perfil normal de cada sensor (factor, desviación respecto a su propia media rolling, mediciones, hora del día). Puntúa cada hora según cuánto se aleja del comportamiento habitual del propio tramo. Contamination = 5 %.
  2. Peer comparison por corredor — para cada hora, compara el factor de un tramo con la media de los otros tramos del mismo itinerario. Si el residuo supera 3σ respecto al error habitual del corredor, se marca como sospechoso. Distingue eventos reales de tráfico (todos los tramos del corredor suben juntos) de fallos de sensor (solo uno diverge).

El flag final sensor_trafico_ok (1 = fiable, 0 = sospechoso) se activa si cualquiera de las tres señales detecta un problema.

import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np
from sklearn.ensemble import IsolationForest

# ── Parámetros ────────────────────────────────────────────────────────────────
NULL_UMBRAL  = 20    # % nulos en un mes × tram → periodo malo
ROLL_WIN     = 24    # ventana rolling para media propia del tram (horas)
IF_CONTAM    = 0.05  # fracción esperada de anomalías en Isolation Forest
PEER_ZSCORE  = 3.0   # desviaciones sobre el error del corredor → sospechoso

trafico_diag = trafico_horario.copy()
trafico_diag["anyo_mes"]   = trafico_diag["hora"].dt.to_period("M")
trafico_diag["hora_dia"]   = trafico_diag["hora"].dt.hour
trafico_diag["dia_semana"] = trafico_diag["hora"].dt.dayofweek
trafico_diag = trafico_diag.sort_values(["idTram", "hora"]).reset_index(drop=True)

# ── 1. Tasa de nulos por mes × tram ──────────────────────────────────────────
null_tram_mes = (
    trafico_diag.groupby(["idTram", "anyo_mes"])["factor_medio"]
    .apply(lambda g: g.isnull().mean() * 100)
    .round(1)
    .reset_index(name="null_pct")
)
null_tram_mes["null_excesivo"] = null_tram_mes["null_pct"] > NULL_UMBRAL

# ── 2. Isolation Forest por tramo ────────────────────────────────────────────
# Features: desviación del factor respecto a la media rolling propia del tramo,
# n_mediciones y contexto temporal. Evita el falso positivo de std=0 sobre 1.0.
trafico_diag["roll_mean"] = (
    trafico_diag.groupby("idTram")["factor_medio"]
    .transform(lambda s: s.rolling(ROLL_WIN, min_periods=6).mean())
)
trafico_diag["desv_roll"] = (
    (trafico_diag["factor_medio"] - trafico_diag["roll_mean"]).abs()
)

feat_if = ["factor_medio", "desv_roll", "n_mediciones", "hora_dia", "dia_semana"]
trafico_diag["if_anomalia"] = 0

for tram_id, grp in trafico_diag.groupby("idTram"):
    mask_ok = grp[feat_if].notna().all(axis=1)
    if mask_ok.sum() < 50:
        continue
    X = grp.loc[mask_ok, feat_if].fillna(0)
    clf = IsolationForest(contamination=IF_CONTAM, random_state=42, n_jobs=-1)
    preds = clf.fit_predict(X)              # -1 = anomalía, 1 = normal
    trafico_diag.loc[X.index, "if_anomalia"] = (preds == -1).astype(int)

n_if = trafico_diag["if_anomalia"].sum()
print(f"Isolation Forest : {n_if:,} horas anómalas "
      f"({n_if / len(trafico_diag) * 100:.1f} %)")

# ── 3. Peer comparison por corredor ──────────────────────────────────────────
# tram_itinerari se carga aquí para que esta sección sea autónoma respecto a la
# Sección 6, donde se define de forma completa con su markdown asociado.
_tram_it_csv = pd.read_csv(
    DATA_DIR / 'Trams - Itineraris' / 'transit_relacio_tram_per_itinerari.csv',
    encoding='latin-1',
)
_tram_it_csv = _tram_it_csv[_tram_it_csv['Tram'] != 0].copy()
tram_itinerari = _tram_it_csv.rename(columns={'Tram': 'idTram', 'Itinerari': 'idItinerari'})[['idTram', 'idItinerari']].drop_duplicates('idTram')
# Media del corredor = media de los otros tramos del mismo itinerario en esa hora.
# Residuo alto y atípico → sospechoso.
tram_it = (
    trafico_diag[["idTram", "hora", "hora_dia", "factor_medio"]]
    .merge(tram_itinerari[["idTram", "idItinerari"]], on="idTram", how="left")
    .dropna(subset=["idItinerari"])
)

corredor_mean = tram_it.groupby(["idItinerari", "hora"])["factor_medio"].transform("mean")
n_corredor    = tram_it.groupby(["idItinerari", "hora"])["factor_medio"].transform("count")

tram_it["peer_residuo"] = np.where(
    n_corredor >= 2,
    (tram_it["factor_medio"] - corredor_mean).abs(),
    np.nan,
)

# Z-score del residuo dentro de cada (corredor, hora del día)
tram_it["peer_z"] = (
    tram_it.groupby(["idItinerari", "hora_dia"])["peer_residuo"]
    .transform(lambda s: (s - s.mean()) / (s.std() + 1e-9))
)
tram_it["peer_anomalia"] = (tram_it["peer_z"] > PEER_ZSCORE).astype(int)

n_peer = tram_it["peer_anomalia"].sum()
print(f"Peer comparison  : {n_peer:,} horas anómalas "
      f"({n_peer / len(tram_it) * 100:.1f} %)")

# ── 4. Flag final: sensor_trafico_ok ─────────────────────────────────────────
trafico_horario = trafico_horario.copy()
trafico_horario["anyo_mes"] = trafico_horario["hora"].dt.to_period("M")

trafico_horario = trafico_horario.merge(
    null_tram_mes[["idTram", "anyo_mes", "null_excesivo"]],
    on=["idTram", "anyo_mes"], how="left"
)
trafico_horario["null_excesivo"] = trafico_horario["null_excesivo"].fillna(False)

trafico_horario = trafico_horario.merge(
    trafico_diag[["idTram", "hora", "if_anomalia"]],
    on=["idTram", "hora"], how="left"
)
trafico_horario["if_anomalia"] = trafico_horario["if_anomalia"].fillna(0).astype(int)

peer_flags = (
    tram_it[["idTram", "hora", "peer_anomalia"]]
    .drop_duplicates(["idTram", "hora"])
)
trafico_horario = trafico_horario.merge(
    peer_flags, on=["idTram", "hora"], how="left"
)
trafico_horario["peer_anomalia"] = trafico_horario["peer_anomalia"].fillna(0).astype(int)

trafico_horario["sensor_trafico_ok"] = (
    ~trafico_horario["null_excesivo"]
    & (trafico_horario["if_anomalia"]   == 0)
    & (trafico_horario["peer_anomalia"] == 0)
).astype(int)

trafico_horario = trafico_horario.drop(
    columns=["anyo_mes", "null_excesivo", "if_anomalia", "peer_anomalia"]
)

n_malas = (trafico_horario["sensor_trafico_ok"] == 0).sum()
print(f"\nHoras sospechosas totales : {n_malas:,} de {len(trafico_horario):,} "
      f"({n_malas / len(trafico_horario) * 100:.1f} %)")
print(f"\nMeses con >{NULL_UMBRAL}% nulos (top 10):")
print(

    .sort_values("null_pct", ascending=False)

    .to_string(index=False)
)

# Interpolación temporal de variables sinópticas antes del promedio de estaciones.
# AEMET solo registra presión/nubosidad/visibilidad en observaciones sinópticas
# (4×/día para presión: 00h, 07h, 13h, 18h; 3×/día para nubosidad y visibilidad: 07h, 13h, 18h).
# El forward-fill por estación propaga el último valor observado hasta la siguiente observación.
_SINOPTICAS = {
    'presion_hpa':     7,   # hueco máx entre observaciones = 7 h (18h → 00h siguiente)
    'nubosidad_oktas': 13,  # hueco máx = 13 h (18h → 07h siguiente)
    'visibilidad_km':  13,
}
_aemet = aemet_raw.sort_values(['estacion', 'fecha']).copy()
for _col, _lim in _SINOPTICAS.items():
    _aemet[_col] = (
        _aemet.groupby('estacion')[_col]
        .transform(lambda x: x.ffill(limit=_lim).bfill(limit=_lim))
    )

    .reset_index()
    .rename(columns={
        'fecha':        'hora',
        'temp_c':       'temp_media_c',
        'humedad_pct':  'humedad_rel_pct',

)

# Expandir eventos a días individuales
def expandir_a_dias(df_eventos, col_inicio, col_fin):
    """Genera un registro por día para cada evento durante su duración."""
    rows = []
    for _, ev in df_eventos.iterrows():
        ini = ev[col_inicio].normalize() if pd.notna(ev[col_inicio]) else None
        fin = ev[col_fin].normalize()    if pd.notna(ev[col_fin])    else ini
        if ini is None:
            continue
        if fin < ini:
            fin = ini
        # Limitar duración máxima a 30 días para evitar explosión de memoria
        fin = min(fin, ini + pd.Timedelta(days=30))
        for dia in pd.date_range(ini, fin, freq='D'):
            rows.append({'fecha': dia, 'tipo_evento': ev.get('tipo_evento', 'evento')})
    return pd.DataFrame(rows)

def agregar_rasgos_temporales(df: pd.DataFrame, col_hora: str = 'hora') -> pd.DataFrame:
    """Añade columnas de rasgos temporales derivados de la columna de hora."""
    t = df[col_hora]
    df = df.copy()

    df['anyo']          = t.dt.year
    df['mes']           = t.dt.month
    df['dia_mes']       = t.dt.day
    df['hora_dia']      = t.dt.hour
    df['dia_semana']    = t.dt.dayofweek          # 0=Lunes
    df['es_fin_semana'] = (df['dia_semana'] >= 5).astype(int)
    df['semana_anyo']   = t.dt.isocalendar().week.astype(int)

    # Codificación cíclica (sin/cos) para hora y mes
    df['hora_sin'] = np.sin(2 * np.pi * df['hora_dia'] / 24)
    df['hora_cos'] = np.cos(2 * np.pi * df['hora_dia'] / 24)
    df['mes_sin']  = np.sin(2 * np.pi * df['mes'] / 12)
    df['mes_cos']  = np.cos(2 * np.pi * df['mes'] / 12)
    df['dia_sin']  = np.sin(2 * np.pi * df['dia_semana'] / 7)
    df['dia_cos']  = np.cos(2 * np.pi * df['dia_semana'] / 7)

    # Franjas horarias
    h = df['hora_dia']
    df['franja_horaria'] = pd.cut(
        h,
        bins=[-1, 6, 10, 14, 17, 20, 23],
        labels=['noche', 'manyana', 'mediodia', 'tarde_temprana', 'tarde', 'noche_tarde']
    )

    return df

def extraer_coordenadas(coord_str):
    """Extrae lat/lon de inicio y fin de la cadena de coordenadas del tram."""
    try:
        parts = [float(x.strip()) for x in str(coord_str).split(',') if x.strip()]
        if len(parts) >= 4:
            return parts[0], parts[1], parts[-2], parts[-1]

        pass
